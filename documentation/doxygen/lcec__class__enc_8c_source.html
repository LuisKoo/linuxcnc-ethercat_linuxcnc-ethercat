52 int class_enc_init(struct lcec_slave *slave, lcec_class_enc_data_t *hal_data, int raw_bits, const char *pfx) {
53  lcec_master_t *master = slave->master;
54  int err;
55 
56  // export pins
57  if ((err = lcec_pin_newf_list(hal_data, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, pfx)) != 0) {
58  return err;
59  }
60 
61  // export parameters
62  if ((err = lcec_param_newf_list(hal_data, slave_params, LCEC_MODULE_NAME, master->name, slave->name, pfx)) != 0) {
63  return err;
64  }
65 
66  hal_data->do_init = 1;
67  hal_data->index_sign = 0;
68 
69  hal_data->pprev_last = 0;
70  hal_data->pprev_scale = 1.0;
71 
72  hal_data->raw_bits = raw_bits;
73  hal_data->raw_shift = 32 - hal_data->raw_bits;
74  hal_data->raw_mask = (1LL << hal_data->raw_bits) - 1;
75 
76  return 0;
77 }

79 void class_enc_update(lcec_class_enc_data_t *hal_data, uint64_t pprev, double scale, uint32_t raw, uint32_t ext_latch_raw, int ext_latch_ena) {
80  long long pos, mod;
81  uint32_t ovfl_win;
82  int sign;
83  double pos_scale;
84 
85  // calculate pos scale
86  if (pprev > 0) {
87  if (hal_data->do_init || hal_data->pprev_last != pprev) {
88  hal_data->pprev_scale = 1.0 / ((double) pprev);
89  hal_data->index_sign = 0;
90  }
91  pos_scale = hal_data->pprev_scale * scale;
92  } else {
93  pos_scale = scale;
94  }
95  hal_data->pprev_last = pprev;
96 
97  // init last encoder value to last known position
98  // this could be used to retain extrapolated multiturn tracking
99  // IMPORTANT: hi/lo values need to be stored atomic
100  if (hal_data->do_init) {
101  *(hal_data->raw) = *(hal_data->ext_lo) & hal_data->raw_mask;
102  }
103 
104  // extrapolate to 64 bits
105  pos = ((long long) *(hal_data->ext_hi) << 32) | *(hal_data->ext_lo);
106  pos += raw_diff(hal_data->raw_shift, raw, *(hal_data->raw));
107  *(hal_data->raw) = raw;
108  *(hal_data->ext_hi) = (uint32_t) (pos >> 32);
109  *(hal_data->ext_lo) = (uint32_t) pos;
110 
111  // set raw encoder pos
112  *(hal_data->pos_enc) = ((double) pos) * pos_scale;
113 
114  // calculate home based abs pos
115  pos += raw_diff(hal_data->raw_shift, 0, hal_data->raw_home);
116  *(hal_data->pos_abs) = ((double) pos) * pos_scale;
117  *(hal_data->on_home_neg) = (pos <= 0);
118  *(hal_data->on_home_pos) = (pos >= 0);
119 
120  // handle index
121  if (*(hal_data->index_ena)) {
122  // get overflow detection window (pprev / 4)
123  ovfl_win = pprev >> 2;
124  if (ovfl_win == 0 || pprev > 0xffffffff) {
125  // no useable singleturn bits -> just reset the position
126  *(hal_data->index_ena) = 0;
127  set_ref(hal_data, pos);
128  } else {
129  mod = signed_mod_64(pos, pprev);
130  sign = (mod >= 0) ? 1 : -1;
131  if (hal_data->index_sign != 0 && sign != hal_data->index_sign && mod <= ovfl_win) {
132  *(hal_data->index_ena) = 0;
133  set_ref(hal_data, pos - mod);
134  }
135  hal_data->index_sign = sign;
136  }
137  } else {
138  hal_data->index_sign = 0;
139  }
140 
141  // handle external latch
142  if (ext_latch_ena) {
143  set_ref(hal_data, pos + raw_diff(hal_data->raw_shift, ext_latch_raw, raw));
144  }
145 
146  // handle rel position init
147  if (hal_data->do_init || *(hal_data->pos_reset)) {
148  set_ref(hal_data, pos);
149  }
150 
151  // calculate rel pos
152  pos -= ((long long) *(hal_data->ref_hi) << 32) | *(hal_data->ref_lo);
153  *(hal_data->pos) = ((double) pos) * pos_scale;
154 
155  hal_data->do_init = 0;
156 }

158 static int32_t raw_diff(int shift, uint32_t a, uint32_t b) {
159  return ((int32_t) (a << shift) - (int32_t) (b << shift)) >> shift;
160 }

167 static long long signed_mod_64(long long val, unsigned long div) {
168  long long rem = lcec_mod_64(val, div);
169 
170  if (rem < 0) {
171  rem += div;
172  }
173 
174  if (rem > (div >> 1)) {
175  rem -= div;
176  }
177 
178  return rem;
179 }
