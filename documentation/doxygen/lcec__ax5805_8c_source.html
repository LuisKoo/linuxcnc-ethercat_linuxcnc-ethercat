94 static int lcec_ax5805_preinit(struct lcec_slave *slave) {
95  lcec_master_t *master = slave->master;
96  struct lcec_slave *ax5n_slave;
97 
98  // try to find corresponding ax5n
99  ax5n_slave = lcec_slave_by_index(master, slave->index - 1);
100  if (ax5n_slave == NULL) {
101  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: Unable to find corresponding AX5nxx with index %d.\n", master->name, slave->name, slave->index - 1);
102  return -EINVAL;
103  }
104 
105  // check for AX5nxx
106  if (ax5n_slave->proc_preinit != lcec_ax5100_preinit && ax5n_slave->proc_preinit != lcec_ax5200_preinit) {
107  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: Slave with index %d is not an AX5nxx.\n", master->name, slave->name, ax5n_slave->index);
108  return -EINVAL;
109  }
110 
111  // call AX52xx preinit to solve dependency
112  ax5n_slave->proc_preinit(ax5n_slave);
113 
114  // use FSOE config from AX5nxx
115  slave->fsoeConf = ax5n_slave->fsoeConf;
116  if (slave->fsoeConf == NULL) {
117  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "%s.%s: Corresponding AX5nxx with index %d has no FSOE config.\n", master->name, slave->name, ax5n_slave->index);
118  return -EINVAL;
119  }
120 
121  // set PDO count
122  slave->pdo_entry_count = 4 + 3 * slave->fsoeConf->data_channels;
123 
124  return 0;
125 }

127 static int lcec_ax5805_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
128  lcec_master_t *master = slave->master;
129  lcec_ax5805_data_t *hal_data;
130  int err;
131  const lcec_pindesc_t *slave_pins;
132 
133  // initialize callbacks
134  slave->proc_read = lcec_ax5805_read;
135 
136  // alloc hal memory
137  if ((hal_data = hal_malloc(sizeof(lcec_ax5805_data_t))) == NULL) {
138  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
139  return -EIO;
140  }
141  memset(hal_data, 0, sizeof(lcec_ax5805_data_t));
142  slave->hal_data = hal_data;
143 
144  // initialize POD entries
145  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE700, 0x01, &hal_data->fsoe_master_cmd_os, NULL);
146  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE700, 0x02, &hal_data->fsoe_master_connid_os, NULL);
147  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE600, 0x01, &hal_data->fsoe_slave_cmd_os, NULL);
148  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE600, 0x02, &hal_data->fsoe_slave_connid_os, NULL);
149 
150  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE700, 0x03, &hal_data->fsoe_master_crc0_os, NULL);
151  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE600, 0x03, &hal_data->fsoe_slave_crc0_os, NULL);
152  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6640, 0x00, &hal_data->fsoe_in_sto0_os, &hal_data->fsoe_in_sto0_bp);
153 
154  if (slave->fsoeConf->data_channels >= 2) {
155  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE700, 0x04, &hal_data->fsoe_master_crc1_os, NULL);
156  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0xE600, 0x04, &hal_data->fsoe_slave_crc1_os, NULL);
157  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6E40, 0x00, &hal_data->fsoe_in_sto1_os, &hal_data->fsoe_in_sto1_bp);
158 
159  slave_pins = slave_pins_2ch;
160  } else {
161  slave_pins = slave_pins_1ch;
162  }
163 
164  // export pins
165  if ((err = lcec_pin_newf_list(hal_data, slave_pins, LCEC_MODULE_NAME, master->name, slave->name)) != 0) {
166  return err;
167  }
168 
169  return 0;
170 }

172 static void lcec_ax5805_read(struct lcec_slave *slave, long period) {
173  lcec_master_t *master = slave->master;
174  lcec_ax5805_data_t *hal_data = (lcec_ax5805_data_t *) slave->hal_data;
175  uint8_t *pd = master->process_data;
176 
177  copy_fsoe_data(slave, hal_data->fsoe_slave_cmd_os, hal_data->fsoe_master_cmd_os);
178 
179  *(hal_data->fsoe_master_cmd) = EC_READ_U8(&pd[hal_data->fsoe_master_cmd_os]);
180  *(hal_data->fsoe_master_connid) = EC_READ_U16(&pd[hal_data->fsoe_master_connid_os]);
181  *(hal_data->fsoe_slave_cmd) = EC_READ_U8(&pd[hal_data->fsoe_slave_cmd_os]);
182  *(hal_data->fsoe_slave_connid) = EC_READ_U16(&pd[hal_data->fsoe_slave_connid_os]);
183 
184  *(hal_data->fsoe_master_crc0) = EC_READ_U16(&pd[hal_data->fsoe_master_crc0_os]);
185  *(hal_data->fsoe_slave_crc0) = EC_READ_U16(&pd[hal_data->fsoe_slave_crc0_os]);
186  *(hal_data->fsoe_in_sto0) = EC_READ_BIT(&pd[hal_data->fsoe_in_sto0_os], hal_data->fsoe_in_sto0_bp);
187 
188  if (slave->fsoeConf->data_channels >= 2) {
189  *(hal_data->fsoe_master_crc1) = EC_READ_U16(&pd[hal_data->fsoe_master_crc1_os]);
190  *(hal_data->fsoe_slave_crc1) = EC_READ_U16(&pd[hal_data->fsoe_slave_crc1_os]);
191  *(hal_data->fsoe_in_sto1) = EC_READ_BIT(&pd[hal_data->fsoe_in_sto1_os], hal_data->fsoe_in_sto1_bp);
192  }
193 }
