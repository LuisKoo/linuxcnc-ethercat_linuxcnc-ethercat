71 static int lcec_el31x4_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
72  lcec_master_t *master = slave->master;
73  lcec_el31x4_data_t *hal_data;
74  lcec_el31x4_chan_t *chan;
75  int i;
76  int err;
77 
78  // initialize callbacks
79  slave->proc_read = lcec_el31x4_read;
80 
81  // alloc hal memory
82  if ((hal_data = hal_malloc(sizeof(lcec_el31x4_data_t))) == NULL) {
83  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
84  return -EIO;
85  }
86  memset(hal_data, 0, sizeof(lcec_el31x4_data_t));
87  slave->hal_data = hal_data;
88 
89  // initialize pins
90  for (i=0; i<LCEC_EL31x4_CHANS; i++) {
91  chan = &hal_data->chans[i];
92 
93  // initialize POD entries
94  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x01, &chan->udr_pdo_os, &chan->udr_pdo_bp);
95  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x02, &chan->ovr_pdo_os, &chan->ovr_pdo_bp);
96  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x07, &chan->error_pdo_os, &chan->error_pdo_bp);
97  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x0E, &chan->sync_err_pdo_os, &chan->sync_err_pdo_bp);
98  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x11, &chan->val_pdo_os, NULL);
99 
100  // export pins
101  if ((err = lcec_pin_newf_list(chan, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
102  return err;
103  }
104 
105  // initialize pins
106  *(chan->scale) = 1.0;
107  }
108 
109  return 0;
110 }

112 static void lcec_el31x4_read(struct lcec_slave *slave, long period) {
113  lcec_master_t *master = slave->master;
114  lcec_el31x4_data_t *hal_data = (lcec_el31x4_data_t *) slave->hal_data;
115  uint8_t *pd = master->process_data;
116  int i;
117  lcec_el31x4_chan_t *chan;
118  int16_t value;
119 
120  // wait for slave to be operational
121  if (!slave->state.operational) {
122  return;
123  }
124 
125  // check inputs
126  for (i=0; i<LCEC_EL31x4_CHANS; i++) {
127  chan = &hal_data->chans[i];
128 
129  // update state
130  // update state
131  *(chan->overrange) = EC_READ_BIT(&pd[chan->ovr_pdo_os], chan->ovr_pdo_bp);
132  *(chan->underrange) = EC_READ_BIT(&pd[chan->udr_pdo_os], chan->udr_pdo_bp);
133  *(chan->error) = EC_READ_BIT(&pd[chan->error_pdo_os], chan->error_pdo_bp);
134  *(chan->sync_err) = EC_READ_BIT(&pd[chan->sync_err_pdo_os], chan->sync_err_pdo_bp);
135 
136  // update value
137  value = EC_READ_S16(&pd[chan->val_pdo_os]);
138  *(chan->raw_val) = value;
139  *(chan->val) = *(chan->bias) + *(chan->scale) * (double)value * ((double)1/(double)0x7fff);
140  }
