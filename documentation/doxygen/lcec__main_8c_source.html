92 int rtapi_app_main(void) {
93  int slave_count;
94  lcec_master_t *master;
95  lcec_slave_t *slave;
96  char name[HAL_NAME_LEN + 1];
97  ec_pdo_entry_reg_t *pdo_entry_regs;
98  lcec_slave_sdoconf_t *sdo_config;
99  lcec_slave_idnconf_t *idn_config;
100  struct timeval tv;
101 
102  // connect to the HAL
103  if ((lcec_comp_id = hal_init(LCEC_MODULE_NAME)) < 0) {
104  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_init() failed\n");
105  goto fail0;
106  }
107 
108  // parse configuration
109  if ((slave_count = lcec_parse_config()) < 0) {
110  goto fail1;
111  }
112 
113  // init global hal data
114  if ((global_hal_data = lcec_init_master_hal(LCEC_MODULE_NAME, 1)) == NULL) {
115  goto fail2;
116  }
117 
118  // initialize masters
119  for (master = first_master; master != NULL; master = master->next) {
120  // request ethercat master
121  if (!(master->master = ecrt_request_master(master->index))) {
122  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "requesting master %s (index %d) failed\n", master->name, master->index);
123  goto fail2;
124  }
125 
126 #ifdef __KERNEL__
127  // register callbacks
128  ecrt_master_callbacks(master->master, lcec_request_lock, lcec_release_lock, master);
129 #endif
130 
131  // create domain
132  if (!(master->domain = ecrt_master_create_domain(master->master))) {
133  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "master %s domain creation failed\n", master->name);
134  goto fail2;
135  }
136 
137  // initialize slaves
138  pdo_entry_regs = master->pdo_entry_regs;
139  for (slave = master->first_slave; slave != NULL; slave = slave->next) {
140  // read slave config
141 
142  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "calling ecrt_master_slave_config for slave %s.%s\n", master->name, slave->name);
143  if (!(slave->config = ecrt_master_slave_config(master->master, 0, slave->index, slave->vid, slave->pid))) {
144  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "fail to read slave %s.%s configuration\n", master->name, slave->name);
145  goto fail2;
146  }
147 
148  // initialize sdos
149  if (slave->sdo_config != NULL) {
150  for (sdo_config = slave->sdo_config; sdo_config->index != 0xffff;
151  sdo_config = (lcec_slave_sdoconf_t *)&sdo_config->data[sdo_config->length]) {
152  if (sdo_config->subindex == LCEC_CONF_SDO_COMPLETE_SUBIDX) {
153  if (ecrt_slave_config_complete_sdo(slave->config, sdo_config->index, &sdo_config->data[0], sdo_config->length) != 0) {
154  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failed to configure slave %s.%s sdo %04x (complete)\n", master->name,
155  slave->name, sdo_config->index);
156  }
157  } else {
158  if (ecrt_slave_config_sdo(slave->config, sdo_config->index, sdo_config->subindex, &sdo_config->data[0], sdo_config->length) !=
159  0) {
160  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failed to configure slave %s.%s sdo %04x:%02x\n", master->name, slave->name,
161  sdo_config->index, sdo_config->subindex);
162  }
163  }
164  }
165  }
166 
167  // initialize idns
168  if (slave->idn_config != NULL) {
169  for (idn_config = slave->idn_config; idn_config->state != 0;
170  idn_config = (lcec_slave_idnconf_t *)&idn_config->data[idn_config->length]) {
171  if (ecrt_slave_config_idn(
172  slave->config, idn_config->drive, idn_config->idn, idn_config->state, &idn_config->data[0], idn_config->length) != 0) {
173  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "fail to configure slave %s.%s drive %d idn %c-%d-%d (state %d, length %u)\n",
174  master->name, slave->name, idn_config->drive, (idn_config->idn & 0x8000) ? 'P' : 'S', (idn_config->idn >> 12) & 0x0007,
175  idn_config->idn & 0x0fff, idn_config->state, (unsigned int)idn_config->length);
176  }
177  }
178  }
179 
180  // setup pdos
181  if (slave->proc_init != NULL) {
182  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "proc_init for slave %s.%s\n", master->name, slave->name);
183  if ((slave->proc_init(lcec_comp_id, slave, pdo_entry_regs)) != 0) {
184  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failure in proc_init for slave %s.%s\n", master->name, slave->name);
185  goto fail2;
186  }
187  }
188  pdo_entry_regs += slave->pdo_entry_count;
189 
190  // configure dc for this slave
191  if (slave->dc_conf != NULL) {
192  ecrt_slave_config_dc(slave->config, slave->dc_conf->assignActivate, slave->dc_conf->sync0Cycle, slave->dc_conf->sync0Shift,
193  slave->dc_conf->sync1Cycle, slave->dc_conf->sync1Shift);
194  rtapi_print_msg(RTAPI_MSG_DBG,
195  LCEC_MSG_PFX "configuring DC for slave %s.%s: assignActivate=x%x sync0Cycle=%d sync0Shift=%d sync1Cycle=%d sync1Shift=%d\n",
196  master->name, slave->name, slave->dc_conf->assignActivate, slave->dc_conf->sync0Cycle, slave->dc_conf->sync0Shift,
197  slave->dc_conf->sync1Cycle, slave->dc_conf->sync1Shift);
198  }
199 
200  // Configure the slave's watchdog times.
201  if (slave->wd_conf != NULL) {
202  ecrt_slave_config_watchdog(slave->config, slave->wd_conf->divider, slave->wd_conf->intervals);
203  }
204 
205  // configure slave
206  if (slave->sync_info != NULL) {
207  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "sync_info setup for slave %s.%s\n", master->name, slave->name);
208  if (ecrt_slave_config_pdos(slave->config, EC_END, slave->sync_info)) {
209  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "fail to configure slave %s.%s\n", master->name, slave->name);
210  goto fail2;
211  }
212  }
213 
214  // export state pins
215  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "init slave state hal for slave %s.%s\n", master->name, slave->name);
216  if ((slave->hal_state_data = lcec_init_slave_state_hal(master->name, slave->name)) == NULL) {
217  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failure to export slave pins for slave %s.%s\n", master->name, slave->name);
218  goto fail2;
219  }
220  }
221 
222  // terminate POD entries
223  pdo_entry_regs->index = 0;
224 
225  // register PDO entries
226  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "register PDO entries\n");
227  if (ecrt_domain_reg_pdo_entry_list(master->domain, master->pdo_entry_regs)) {
228  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "master %s PDO entry registration failed\n", master->name);
229  goto fail2;
230  }
231 
232  // initialize application time
233  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "Setting time\n");
234  lcec_gettimeofday(&tv);
235  master->app_time_base = EC_TIMEVAL2NANO(tv);
236  ecrt_master_application_time(master->master, master->app_time_base);
237 #ifdef RTAPI_TASK_PLL_SUPPORT
238  master->dc_time_valid_last = 0;
239  if (master->sync_ref_cycles >= 0) {
240  master->app_time_base -= rtapi_get_time();
241  }
242 #else
243  master->app_time_base -= rtapi_get_time();
244  if (master->sync_ref_cycles < 0) {
245  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "unable to sync master %s cycle to reference clock, RTAPI_TASK_PLL_SUPPORT not present\n",
246  master->name);
247  }
248 #endif
249 
250  // activating master
251  rtapi_print_msg(RTAPI_MSG_DBG, LCEC_MSG_PFX "Activating master\n");
252  if (ecrt_master_activate(master->master)) {
253  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failed to activate master %s\n", master->name);
254  goto fail2;
255  }
256 
257  // Get internal process data for domain
258  master->process_data = ecrt_domain_data(master->domain);
259  master->process_data_len = ecrt_domain_size(master->domain);
260 
261  // init hal data
262  rtapi_snprintf(name, HAL_NAME_LEN, "%s.%s", LCEC_MODULE_NAME, master->name);
263  if ((master->hal_data = lcec_init_master_hal(name, 0)) == NULL) {
264  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failure to init hal pins for slave %s.%s\n", master->name, slave->name);
265  goto fail2;
266  }
267 
268 #ifdef RTAPI_TASK_PLL_SUPPORT
269  // set default PLL_STEP: use +/-0.1% of period
270  master->hal_data->pll_step = master->app_time_period / 1000;
271  // set default PLL_MAX_ERR: one period
272  master->hal_data->pll_max_err = master->app_time_period;
273 #endif
274 
275  // export read function
276  rtapi_snprintf(name, HAL_NAME_LEN, "%s.%s.read", LCEC_MODULE_NAME, master->name);
277  if (hal_export_funct(name, lcec_read_master, master, 0, 0, lcec_comp_id) != 0) {
278  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "master %s read funct export failed\n", master->name);
279  goto fail2;
280  }
281  // export write function
282  rtapi_snprintf(name, HAL_NAME_LEN, "%s.%s.write", LCEC_MODULE_NAME, master->name);
283  if (hal_export_funct(name, lcec_write_master, master, 0, 0, lcec_comp_id) != 0) {
284  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "master %s write funct export failed\n", master->name);
285  goto fail2;
286  }
287  }
288 
289  // export read-all function
290  rtapi_snprintf(name, HAL_NAME_LEN, "%s.read-all", LCEC_MODULE_NAME);
291  if (hal_export_funct(name, lcec_read_all, NULL, 0, 0, lcec_comp_id) != 0) {
292  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "read-all funct export failed\n");
293  goto fail2;
294  }
295  // export write-all function
296  rtapi_snprintf(name, HAL_NAME_LEN, "%s.write-all", LCEC_MODULE_NAME);
297  if (hal_export_funct(name, lcec_write_all, NULL, 0, 0, lcec_comp_id) != 0) {
298  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "write-all funct export failed\n");
299  goto fail2;
300  }
301 
302  rtapi_print_msg(RTAPI_MSG_INFO, LCEC_MSG_PFX "installed driver for %d slaves\n", slave_count);
303  hal_ready(lcec_comp_id);
304  return 0;
305 
306 fail2:
307  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "failure, clearing config\n");
308  lcec_clear_config();
309 fail1:
310  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "exiting\n");
311  hal_exit(lcec_comp_id);
312 fail0:
313  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "returning -EINVAL\n");
314  return -EINVAL;
315 }

318  lcec_master_t *master;
319 
320  // deactivate all masters
321  for (master = first_master; master != NULL; master = master->next) {
322  ecrt_master_deactivate(master->master);
323  }
324 
325  lcec_clear_config();
326  hal_exit(lcec_comp_id);
327 }

329 int lcec_parse_config(void) {
330  int shmem_id;
331  void *shmem_ptr;
332  LCEC_CONF_HEADER_T *header;
333  size_t length;
334  void *conf;
335  int slave_count;
336  const lcec_typelist_t *type;
337  lcec_master_t *master;
338  lcec_slave_t *slave;
339  lcec_slave_dc_t *dc;
340  lcec_slave_watchdog_t *wd;
341  ec_pdo_entry_reg_t *pdo_entry_regs;
342  LCEC_CONF_TYPE_T conf_type;
343  LCEC_CONF_MASTER_T *master_conf;
344  LCEC_CONF_SLAVE_T *slave_conf;
345  LCEC_CONF_DC_T *dc_conf;
346  LCEC_CONF_WATCHDOG_T *wd_conf;
347  LCEC_CONF_SYNCMANAGER_T *sm_conf;
348  LCEC_CONF_PDO_T *pdo_conf;
349  LCEC_CONF_PDOENTRY_T *pe_conf;
350  LCEC_CONF_COMPLEXENTRY_T *ce_conf;
351  LCEC_CONF_SDOCONF_T *sdo_conf;
352  LCEC_CONF_IDNCONF_T *idn_conf;
353  LCEC_CONF_MODPARAM_T *modparam_conf;
354  ec_pdo_entry_info_t *generic_pdo_entries;
355  ec_pdo_info_t *generic_pdos;
356  ec_sync_info_t *generic_sync_managers;
357  lcec_generic_pin_t *generic_hal_data;
358  hal_pin_dir_t generic_hal_dir;
359  lcec_slave_sdoconf_t *sdo_config;
360  lcec_slave_idnconf_t *idn_config;
361  lcec_slave_modparam_t *modparams;
362 
363  // initialize list
364  first_master = NULL;
365  last_master = NULL;
366 
367  // try to get config header
368  shmem_id = rtapi_shmem_new(LCEC_CONF_SHMEM_KEY, lcec_comp_id, sizeof(LCEC_CONF_HEADER_T));
369  if (shmem_id < 0) {
370  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "couldn't allocate user/RT shared memory\n");
371  goto fail0;
372  }
373  if (lcec_rtapi_shmem_getptr(shmem_id, &shmem_ptr) < 0) {
374  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "couldn't map user/RT shared memory\n");
375  goto fail1;
376  }
377 
378  // check magic, get length and close shmem
379  header = shmem_ptr;
380  if (header->magic != LCEC_CONF_SHMEM_MAGIC) {
381  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "lcec_conf is not loaded\n");
382  goto fail1;
383  }
384  length = header->length;
385  rtapi_shmem_delete(shmem_id, lcec_comp_id);
386 
387  // reopen shmem with proper size
388  shmem_id = rtapi_shmem_new(LCEC_CONF_SHMEM_KEY, lcec_comp_id, sizeof(LCEC_CONF_HEADER_T) + length);
389  if (shmem_id < 0) {
390  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "couldn't allocate user/RT shared memory\n");
391  goto fail0;
392  }
393  if (lcec_rtapi_shmem_getptr(shmem_id, &shmem_ptr) < 0) {
394  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "couldn't map user/RT shared memory\n");
395  goto fail1;
396  }
397 
398  // get pointer to config
399  conf = shmem_ptr + sizeof(LCEC_CONF_HEADER_T);
400 
401  // process config items
402  slave_count = 0;
403  master = NULL;
404  slave = NULL;
405  generic_pdo_entries = NULL;
406  generic_pdos = NULL;
407  generic_sync_managers = NULL;
408  generic_hal_data = NULL;
409  generic_hal_dir = 0;
410  sdo_config = NULL;
411  idn_config = NULL;
412  pe_conf = NULL;
413  modparams = NULL;
414  while ((conf_type = ((LCEC_CONF_NULL_T *)conf)->confType) != lcecConfTypeNone) {
415  // get type
416  switch (conf_type) {
417  case lcecConfTypeMaster:
418  // get config token
419  master_conf = (LCEC_CONF_MASTER_T *)conf;
420  conf += sizeof(LCEC_CONF_MASTER_T);
421 
422  // alloc master memory
423  master = lcec_zalloc(sizeof(lcec_master_t));
424  if (master == NULL) {
425  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "Unable to allocate master %d structure memory\n", master_conf->index);
426  goto fail2;
427  }
428 
429  // initialize master
430  master->index = master_conf->index;
431  strncpy(master->name, master_conf->name, LCEC_CONF_STR_MAXLEN);
432  master->name[LCEC_CONF_STR_MAXLEN - 1] = 0;
433  master->app_time_period = master_conf->appTimePeriod;
434  master->sync_ref_cycles = master_conf->refClockSyncCycles;
435 
436  // add master to list
437  LCEC_LIST_APPEND(first_master, last_master, master);
438  break;
439 
440  case lcecConfTypeSlave:
441  // get config token
442  slave_conf = (LCEC_CONF_SLAVE_T *)conf;
443  conf += sizeof(LCEC_CONF_SLAVE_T);
444 
445  // check for master
446  if (master == NULL) {
447  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "Master node for slave missing\n");
448  goto fail2;
449  }
450 
451  // check for valid slave type
452  if (!strcmp(slave_conf->typename, "generic")) {
453  type = NULL;
454  } else {
455  type = lcec_findslavetype(slave_conf->typename);
456 
457  if (type == NULL) {
458  rtapi_print_msg(RTAPI_MSG_WARN, LCEC_MSG_PFX "Invalid slave name \"%s\"\n", slave_conf->typename);
459  continue;
460  }
461  }
462 
463  // create new slave
464  slave = lcec_zalloc(sizeof(lcec_slave_t));
465  if (slave == NULL) {
466  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "Unabl eto allocate slave %s.%s structure memory\n", master->name, slave_conf->name);
467  goto fail2;
468  }
469 
470  // initialize slave
471  generic_pdo_entries = NULL;
472  generic_pdos = NULL;
473  generic_sync_managers = NULL;
474  generic_hal_data = NULL;
475  generic_hal_dir = 0;
476  sdo_config = NULL;

907  }
908  }

921  }
922  master->pdo_entry_regs = pdo_entry_regs;
923  }
924 
925  return slave_count;
926 
927 fail2:
928  lcec_clear_config();
929 fail1:
930  rtapi_shmem_delete(shmem_id, lcec_comp_id);
931 fail0:

933 }

935 void lcec_clear_config(void) {
936  lcec_master_t *master, *prev_master;
937  lcec_slave_t *slave, *prev_slave;
938 
939  // iterate all masters
940  master = last_master;
941  while (master != NULL) {
942  prev_master = master->prev;
943 
944  // iterate all masters
945  slave = master->last_slave;
946  while (slave != NULL) {
947  prev_slave = slave->prev;
948 
949  // cleanup slave
950  if (slave->proc_cleanup != NULL) {
951  slave->proc_cleanup(slave);
952  }
953 
954  // free slave
955  if (slave->modparams != NULL) {
956  lcec_free(slave->modparams);
957  }
958  if (slave->sdo_config != NULL) {
959  lcec_free(slave->sdo_config);
960  }
961  if (slave->idn_config != NULL) {
962  lcec_free(slave->idn_config);
963  }
964  if (slave->generic_pdo_entries != NULL) {
965  lcec_free(slave->generic_pdo_entries);
966  }
967  if (slave->generic_pdos != NULL) {
968  lcec_free(slave->generic_pdos);
969  }
970  if (slave->generic_sync_managers != NULL) {
971  lcec_free(slave->generic_sync_managers);
972  }
973  if (slave->dc_conf != NULL) {
974  lcec_free(slave->dc_conf);
975  }
976  if (slave->wd_conf != NULL) {
977  lcec_free(slave->wd_conf);
978  }
979  lcec_free(slave);
980  slave = prev_slave;
981  }
982 
983  // release master
984  if (master->master) {
985  ecrt_release_master(master->master);
986  }
987 
988  // free PDO entry memory
989  if (master->pdo_entry_regs != NULL) {
990  lcec_free(master->pdo_entry_regs);
991  }
992 
993  // free master
994  lcec_free(master);
995  master = prev_master;
996  }

1053 void lcec_update_master_hal(lcec_master_data_t *hal_data, ec_master_state_t *ms) {

1098 void lcec_read_master(void *arg, long period) {
1099  lcec_master_t *master = (lcec_master_t *)arg;
1100  lcec_slave_t *slave;
1101  int check_states;
1102 
1103  // check period
1104  if (period != master->period_last) {
1105  master->period_last = period;
1106  if (master->app_time_period != period) {
1107  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "Invalid appTimePeriod of %u for master %s (should be %ld).\n", master->app_time_period,
1108  master->name, period);
1109  }
1110  }
1111 
1112  // get state check flag
1113  if (master->state_update_timer > 0) {
1114  check_states = 0;
1115  master->state_update_timer -= period;
1116  } else {
1117  check_states = 1;
1118  master->state_update_timer = LCEC_STATE_UPDATE_PERIOD;
1119  }
1120 
1121  // receive process data & master state
1122  rtapi_mutex_get(&master->mutex);
1123  ecrt_master_receive(master->master);
1124  ecrt_domain_process(master->domain);
1125  if (check_states) {
1126  ecrt_master_state(master->master, &master->ms);
1127  }
1128  rtapi_mutex_give(&master->mutex);
1129 
1130  // update state pins
1131  lcec_update_master_hal(master->hal_data, &master->ms);
1132

1134  global_ms.slaves_responding += master->ms.slaves_responding;
1135  global_ms.al_states |= master->ms.al_states;
1136  global_ms.link_up = global_ms.link_up && master->ms.link_up;

1140  // get slaves state
1141  rtapi_mutex_get(&master->mutex);
1142  if (check_states) {
1143  ecrt_slave_config_state(slave->config, &slave->state);
1144  }
1145  rtapi_mutex_give(&master->mutex);

1148  }

1157 void lcec_write_master(void *arg, long period) {
1158  lcec_master_t *master = (lcec_master_t *)arg;
1159  lcec_slave_t *slave;
1160  uint64_t app_time;
1161  long long now;
1162 #ifdef RTAPI_TASK_PLL_SUPPORT
1163  long long ref;
1164  uint32_t dc_time;
1165  int dc_time_valid;
1166  lcec_master_data_t *hal_data;
1167 #endif
1168 
1169  // process slaves
1170  for (slave = master->first_slave; slave != NULL; slave = slave->next) {
1171  if (slave->proc_write != NULL) {
1172  slave->proc_write(slave, period);
1173  }
1174  }
1175 
1176 #ifdef RTAPI_TASK_PLL_SUPPORT
1177  // get reference time
1178  ref = rtapi_task_pll_get_reference();
1179 #endif
1180 
1181  // send process data
1182  rtapi_mutex_get(&master->mutex);
1183  ecrt_domain_queue(master->domain);
1184 
1185  // update application time
1186  now = rtapi_get_time();
1187 #ifdef RTAPI_TASK_PLL_SUPPORT
1188  if (master->sync_ref_cycles >= 0) {
1189  app_time = master->app_time_base + now;
1190  } else {
1191  master->dc_ref += period;
1192  app_time = master->app_time_base + master->dc_ref + (now - ref);
1193  }
1194 #else
1195  app_time = master->app_time_base + now;
1196 #endif
1197 
1198  ecrt_master_application_time(master->master, app_time);
1199 
1200  // sync ref clock to master
1201  if (master->sync_ref_cycles > 0) {
1202  if (master->sync_ref_cnt == 0) {
1203  master->sync_ref_cnt = master->sync_ref_cycles;
1204  ecrt_master_sync_reference_clock(master->master);
1205  }
1206  master->sync_ref_cnt--;
1207  }
1208 
1209 #ifdef RTAPI_TASK_PLL_SUPPORT
1210  // sync master to ref clock
1211  dc_time = 0;
1212  if (master->sync_ref_cycles < 0) {
1213  // get reference clock time to synchronize master cycle
1214  dc_time_valid = (ecrt_master_reference_clock_time(master->master, &dc_time) == 0);
1215  } else {
1216  dc_time_valid = 0;
1217  }
1218 #endif
1219 
1220  // sync slaves to ref clock
1221  ecrt_master_sync_slave_clocks(master->master);
1222 
1223  // send domain data
1224  ecrt_master_send(master->master);
1225  rtapi_mutex_give(&master->mutex);
1226 
1227 #ifdef RTAPI_TASK_PLL_SUPPORT
1228  // BANG-BANG controller for master thread PLL sync
1229  // this part is done after ecrt_master_send() to reduce jitter
1230  hal_data = master->hal_data;
1231  *(hal_data->pll_err) = 0;
1232  *(hal_data->pll_out) = 0;
1233  // the first read dc_time value semms to be invalid, so wait for two successive succesfull reads
1234  if (dc_time_valid && master->dc_time_valid_last) {
1235  *(hal_data->pll_err) = master->app_time_last - dc_time;
1236  // check for invalid error values
1237  if (abs(*(hal_data->pll_err)) > hal_data->pll_max_err) {
1238  // force resync of master time
1239  master->dc_ref -= *(hal_data->pll_err);
1240  // skip next control cycle to allow resync
1241  dc_time_valid = 0;
1242  // increment reset counter to document this event
1243  (*(hal_data->pll_reset_cnt))++;
1244  } else {
1245  *(hal_data->pll_out) = (*(hal_data->pll_err) < 0) ? -(hal_data->pll_step) : (hal_data->pll_step);
1246  }
1247  }
1248 
1249  rtapi_task_pll_set_correction(*(hal_data->pll_out));
1250  master->app_time_last = (uint32_t)app_time;
1251  master->dc_time_valid_last = dc_time_valid;
1252 #endif
1253 }
