34 #define LCEC_LIST_APPEND(first, last, item) \
35  do { \
36  (item)->prev = (last); \
37  if ((item)->prev != NULL) { \
38  (item)->prev->next = (item); \
39  } else { \
40  (first) = (item); \
41  } \
42  (last) = (item); \
43  } while (0);
