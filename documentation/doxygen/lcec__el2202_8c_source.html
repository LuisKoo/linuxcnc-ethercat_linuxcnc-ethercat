93 static int lcec_el2202_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
94  lcec_master_t *master = slave->master;
95 
96  lcec_el2202_data_t *hal_data;
97  lcec_el2202_chan_t *chan;
98 
99  int i;
100  int err;
101 
102  // initialize callbacks
103  slave->proc_write = lcec_el2202_write;
104 
105  // alloc hal memory
106  if ((hal_data = hal_malloc(sizeof(lcec_el2202_data_t))) == NULL) {
107  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
108  return -EIO;
109  }
110  memset(hal_data, 0, sizeof(lcec_el2202_data_t));
111  slave->hal_data = hal_data;
112 
113  // initializer sync info
114  slave->sync_info = lcec_el2202_syncs;
115 
116  // initialize pins
117  for (i=0; i<LCEC_EL2202_CHANS; i++) {
118  chan = &hal_data->chans[i];
119 
120  // initialize PDO entries position vend.id prod.code index sindx offset bit pos
121  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7000 + (i << 4), 0x01, &chan->out_offs, &chan->out_bitp);
122  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7000 + (i << 4), 0x02, &chan->tristate_offs, &chan->tristate_bitp);
123 
124  // export pins
125  if ((err = lcec_pin_newf_list(chan, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
126  return err;
127  }
128  }
129 
130  return 0;
131 }

133 static void lcec_el2202_write(struct lcec_slave *slave, long period) {
134  lcec_master_t *master = slave->master;
135  uint8_t *pd = master->process_data;
136 
137  lcec_el2202_data_t *hal_data = (lcec_el2202_data_t *) slave->hal_data;
138  lcec_el2202_chan_t *chan;
139 
140  int i;
141 
142  for (i=0; i<LCEC_EL2202_CHANS; i++) {
143  chan = &hal_data->chans[i];
144 
145  // set output
146  EC_WRITE_BIT(&pd[chan->out_offs], chan->out_bitp, *(chan->out));
147  // set tristate
148  EC_WRITE_BIT(&pd[chan->tristate_offs], chan->tristate_bitp, *(chan->tristate));
149  }
150 }
