129 static int lcec_el5032_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
130  lcec_master_t *master = slave->master;
131  lcec_el5032_data_t *hal_data;
132  int i;
133  lcec_el5032_chan_t *chan;
134  int err;
135 
136  // initialize callbacks
137  slave->proc_read = lcec_el5032_read;
138 
139  // alloc hal memory
140  if ((hal_data = hal_malloc(sizeof(lcec_el5032_data_t))) == NULL) {
141  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
142  return -EIO;
143  }
144  memset(hal_data, 0, sizeof(lcec_el5032_data_t));
145  slave->hal_data = hal_data;
146 
147  // initialize sync info
148  slave->sync_info = lcec_el5032_syncs;
149 
150  // initialize global data
151  hal_data->last_operational = 0;
152 
153  // initialize pins
154  for (i=0; i<LCEC_EL5032_CHANS; i++) {
155  chan = &hal_data->chans[i];
156 
157  // initialize POD entries
158  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x01, &chan->warn_os, &chan->warn_bp);
159  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x02, &chan->error_os, &chan->error_bp);
160  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x03, &chan->ready_os, &chan->ready_bp);
161  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x0d, &chan->diag_os, &chan->diag_bp);
162  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x0e, &chan->tx_state_os, &chan->tx_state_bp);
163  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x0f, &chan->cyc_cnt_os, &chan->cyc_cnt_bp);
164  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x11, &chan->count_pdo_os, NULL);
165 
166  // export pins
167  if ((err = lcec_pin_newf_list(chan, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
168  return err;
169  }
170 
171  // initialize pins
172  *(chan->pos_scale) = 1.0;
173 
174  // initialize variables
175  chan->do_init = 1;
176  chan->last_count = 0;
177  chan->old_scale = *(chan->pos_scale) + 1.0;
178  chan->scale = 1.0;
179  }
180 
181  return 0;
182 }

184 static void lcec_el5032_read(struct lcec_slave *slave, long period) {
185  lcec_master_t *master = slave->master;
186  lcec_el5032_data_t *hal_data = (lcec_el5032_data_t *) slave->hal_data;
187  uint8_t *pd = master->process_data;
188  int i;
189  lcec_el5032_chan_t *chan;
190  int64_t raw_count, raw_delta;
191 
192  // wait for slave to be operational
193  if (!slave->state.operational) {
194  hal_data->last_operational = 0;
195  return;
196  }
197 
198  // check inputs
199  for (i=0; i<LCEC_EL5032_CHANS; i++) {
200  chan = &hal_data->chans[i];
201 
202  // check for change in scale value
203  if (*(chan->pos_scale) != chan->old_scale) {
204  // scale value has changed, test and update it
205  if ((*(chan->pos_scale) < 1e-20) && (*(chan->pos_scale) > -1e-20)) {
206  // value too small, divide by zero is a bad thing
207  *(chan->pos_scale) = 1.0;
208  }
209  // save new scale to detect future changes
210  chan->old_scale = *(chan->pos_scale);
211  // we actually want the reciprocal
212  chan->scale = 1.0 / *(chan->pos_scale);
213  }
214 
215  // get bit states
216  *(chan->warn) = EC_READ_BIT(&pd[chan->warn_os], chan->warn_bp);
217  *(chan->error) = EC_READ_BIT(&pd[chan->error_os], chan->error_bp);
218  *(chan->ready) = EC_READ_BIT(&pd[chan->ready_os], chan->ready_bp);
219  *(chan->diag) = EC_READ_BIT(&pd[chan->diag_os], chan->diag_bp);
220  *(chan->tx_state) = EC_READ_BIT(&pd[chan->tx_state_os], chan->tx_state_bp);
221 
222  // get cycle counter
223  *(chan->cyc_cnt) = (EC_READ_U8(&pd[chan->cyc_cnt_os]) >> chan->cyc_cnt_bp) & 0x03;
224 
225  // read raw values
226  raw_count = EC_READ_S64(&pd[chan->count_pdo_os]);
227 
228  // check for operational change of slave
229  if (!hal_data->last_operational) {
230  chan->last_count = raw_count;
231  }
232 
233  // update raw values
234  *(chan->raw_count_lo) = raw_count;
235  *(chan->raw_count_hi) = raw_count >> 32;
236 
237  // handle initialization
238  if (chan->do_init || *(chan->reset)) {
239  chan->do_init = 0;
240  chan->last_count = raw_count;
241  *(chan->count) = 0;
242  }
243 
244  // compute net counts
245  raw_delta = raw_count - chan->last_count;
246  chan->last_count = raw_count;
247  *(chan->count) += raw_delta;
248 
249  // scale count to make floating point position
250  if (*(chan->abs_mode)) {
251  *(chan->pos) = raw_count * chan->scale;
252  } else {
253  *(chan->pos) = *(chan->count) * chan->scale;
254  }
255  }
256 
257  hal_data->last_operational = 1;
258 }
