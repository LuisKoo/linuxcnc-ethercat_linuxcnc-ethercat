158 static int lcec_el3255_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
159  lcec_master_t *master = slave->master;
160  lcec_el3255_data_t *hal_data;
161  lcec_el3255_chan_t *chan;
162  int i;
163  int err;
164 
165  // initialize callbacks
166  slave->proc_read = lcec_el3255_read;
167 
168  // alloc hal memory
169  if ((hal_data = hal_malloc(sizeof(lcec_el3255_data_t))) == NULL) {
170  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
171  return -EIO;
172  }
173  memset(hal_data, 0, sizeof(lcec_el3255_data_t));
174  slave->hal_data = hal_data;
175 
176  // initialize sync info
177  slave->sync_info = lcec_el3255_syncs;
178 
179  // initialize pins
180  for (i=0; i<LCEC_EL3255_CHANS; i++) {
181  chan = &hal_data->chans[i];
182 
183  // initialize POD entries
184  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x01, &chan->ovr_pdo_os, &chan->ovr_pdo_bp);
185  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x02, &chan->udr_pdo_os, &chan->udr_pdo_bp);
186  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x07, &chan->error_pdo_os, &chan->error_pdo_bp);
187  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x0e, &chan->sync_err_pdo_os, &chan->sync_err_pdo_bp);
188  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000 + (i << 4), 0x11, &chan->val_pdo_os, NULL);
189 
190  // export pins
191  if ((err = lcec_pin_newf_list(chan, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
192  return err;
193  }
194 
195  // initialize pins
196  *(chan->scale) = 1.0;
197  }
198 
199  return 0;
200 }

202 static void lcec_el3255_read(struct lcec_slave *slave, long period) {
203  lcec_master_t *master = slave->master;
204  lcec_el3255_data_t *hal_data = (lcec_el3255_data_t *) slave->hal_data;
205  uint8_t *pd = master->process_data;
206  int i;
207  lcec_el3255_chan_t *chan;
208  int16_t value;
209 
210  // wait for slave to be operational
211  if (!slave->state.operational) {
212  return;
213  }
214 
215  // check inputs
216  for (i=0; i<LCEC_EL3255_CHANS; i++) {
217  chan = &hal_data->chans[i];
218 
219  // update state
220  *(chan->overrange) = EC_READ_BIT(&pd[chan->ovr_pdo_os], chan->ovr_pdo_bp);
221  *(chan->underrange) = EC_READ_BIT(&pd[chan->udr_pdo_os], chan->udr_pdo_bp);
222  *(chan->error) = EC_READ_BIT(&pd[chan->error_pdo_os], chan->error_pdo_bp);
223  *(chan->sync_err) = EC_READ_BIT(&pd[chan->sync_err_pdo_os], chan->sync_err_pdo_bp);
224 
225  // update value
226  value = EC_READ_S16(&pd[chan->val_pdo_os]);
227  *(chan->raw_val) = value;
228  *(chan->val) = *(chan->bias) + *(chan->scale) * (double)value * ((double)1/(double)0x7fff);
229  }
230 }
