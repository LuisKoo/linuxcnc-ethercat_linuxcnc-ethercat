308 static int lcec_el7041_init(int comp_id, struct lcec_slave *s, ec_pdo_entry_reg_t *r) {
309  lcec_master_t *m = s->master;
310  lcec_el7041_data_t *hd;
311  int err;
312 
313  // initialize callbacks
314  s->proc_read = lcec_el7041_read;
315  s->proc_write = lcec_el7041_write;
316 
317  // alloc hal memory
318  if ((hd = hal_malloc(sizeof(lcec_el7041_data_t))) == NULL) {
319  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", m->name, s->name);
320  return -EIO;
321  }
322  memset(hd, 0, sizeof(lcec_el7041_data_t));
323  s->hal_data = hd;
324 
325  // initialize sync info
326  s->sync_info = lcec_el7041_syncs;
327 
328  // initialize global data
329  hd->last_operational = 0;
330 
331  // initialize PDO entries
332  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7000, 0x01, &hd->ena_latch_c_pdo_os, &hd->ena_latch_c_pdo_bp);
333  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7000, 0x02, &hd->ena_latch_ext_pos_pdo_os, &hd->ena_latch_ext_pos_pdo_bp);
334  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7000, 0x03, &hd->set_count_pdo_os, &hd->set_count_pdo_bp);
335  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7000, 0x04, &hd->ena_latch_ext_neg_pdo_os, &hd->ena_latch_ext_neg_pdo_bp);
336  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7000, 0x11, &hd->set_count_val_pdo_os, NULL);
337 
338  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7010, 0x01, &hd->dcm_ena_pdo_os, &hd->dcm_ena_pdo_bp);
339  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7010, 0x02, &hd->dcm_reset_pdo_os, &hd->dcm_reset_pdo_bp);
340  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7010, 0x03, &hd->dcm_reduce_torque_pdo_os, &hd->dcm_reduce_torque_pdo_bp);
341 
342  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x7010, 0x21, &hd->dcm_velo_pdo_os, NULL);
343 
344  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x01, &hd->latch_c_valid_pdo_os, &hd->latch_c_valid_pdo_bp);
345  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x02, &hd->latch_ext_valid_pdo_os, &hd->latch_ext_valid_pdo_bp);
346  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x03, &hd->set_count_done_pdo_os, &hd->set_count_done_pdo_bp);
347  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x04, &hd->count_underflow_pdo_os, &hd->count_overflow_pdo_bp);
348  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x05, &hd->count_overflow_pdo_os, &hd->count_underflow_pdo_bp);
349  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x08, &hd->expol_stall_pdo_os, &hd->expol_stall_pdo_bp);
350  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x09, &hd->ina_pdo_os, &hd->ina_pdo_bp);
351  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x0a, &hd->inb_pdo_os, &hd->inb_pdo_bp);
352  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x0b, &hd->inc_pdo_os, &hd->inc_pdo_bp);
353  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x0d, &hd->inext_pdo_os, &hd->inext_pdo_bp);
354  LCEC_PDO_INIT(r, s->index, s->vid, s->pid, 0x6000, 0x0e, &hd->sync_err_pdo_os, &hd->sync_err_pdo_bp);

403 static void lcec_el7041_read(struct lcec_slave *s, long period) {

406  uint8_t *pd = m->process_data;
407  int16_t raw_count, raw_latch, raw_delta;
408 
409  // wait for slave to be operational
410  if (!s->state.operational) {
411  hd->last_operational = 0;
412  return;
413  }
414 
415  // check inputs
416 
417  // check for change in scale value
418  if (*(hd->pos_scale) != hd->enc_old_scale) {
419  // scale value has changed, test and update it
420  if ((*(hd->pos_scale) < 1e-20) && (*(hd->pos_scale) > -1e-20)) {
421  // value too small, divide by zero is a bad thing
422  *(hd->pos_scale) = 1.0;

453  *(hd->dcm_tx_toggle) = EC_READ_BIT(&pd[hd->dcm_tx_toggle_pdo_os], hd->dcm_tx_toggle_pdo_bp);
454 
455  hd->internal_fault = *(hd->dcm_error);
456 
457  // read raw values
458  raw_count = EC_READ_S16(&pd[hd->count_pdo_os]);
459  raw_latch = EC_READ_S16(&pd[hd->latch_pdo_os]);
460 
461  // check for operational change of slave
462  if (!hd->last_operational) {
463  hd->enc_last_count = raw_count;
464  }
465 
466  // check for counter set done
467  if (EC_READ_BIT(&pd[hd->set_count_done_pdo_os], hd->set_count_done_pdo_bp)) {
468  hd->enc_last_count = raw_count;
469  *(hd->set_raw_count) = 0;
470  }
471 
472  // update raw values
473  if (!*(hd->set_raw_count)) {
474  *(hd->raw_count) = raw_count;
475  }
476 
477  // handle initialization
478  if (hd->enc_do_init || *(hd->reset)) {
479  hd->enc_do_init = 0;
480  hd->enc_last_count = raw_count;
481  *(hd->count) = 0;
482  }
483 
484  // clear pending fault reset if no fault
485  if (!hd->internal_fault) {
486  hd->fault_reset_retry = 0;
487  }
488 
489  // generate gated fault
490  if (hd->fault_reset_retry > 0) {
491  if (hd->fault_reset_cycle < 1) {
492  hd->fault_reset_cycle++;
493  } else {
494  hd->fault_reset_cycle = 0;
495  hd->fault_reset_state = !hd->fault_reset_state;
496  if (hd->fault_reset_state) {
497  hd->fault_reset_retry--;
498  }
499  }
500  *(hd->fault) = 0;
501  } else {
502  *(hd->fault) = hd->internal_fault;
503  }
504 
505  // handle index
506  if (*(hd->latch_ext_valid)) {
507  *(hd->raw_latch) = raw_latch;
508  hd->enc_last_count = raw_latch;
509  *(hd->count) = 0;
510  *(hd->ena_latch_ext_pos) = 0;
511  *(hd->ena_latch_ext_neg) = 0;
512  }
513 
514  // compute net counts
515  raw_delta = raw_count - hd->enc_last_count;
516  hd->enc_last_count = raw_count;
517  *(hd->count) += raw_delta;
518 
519  // scale count to make floating point position
520  *(hd->pos) = *(hd->count) * hd->enc_scale_recip;
521 
522  hd->last_operational = 1;
523 }

526  lcec_master_t *m = s->master;
527  lcec_el7041_data_t *hd = (lcec_el7041_data_t *)s->hal_data;
528  uint8_t *pd = m->process_data;
529  double tmpval, tmpdc, raw_val;
530  int enable_on_edge;
531 
532  // set outputs
533 
534  // check for enable edge
535  enable_on_edge = *(hd->dcm_enable) && !hd->last_dcm_enable;
536  hd->last_dcm_enable = *(hd->dcm_enable);
537 
538  // check for autoreset
539  if (enable_on_edge && hd->internal_fault) {
540  hd->fault_reset_retry = 1;
541  hd->fault_reset_state = 1;
542  hd->fault_reset_cycle = 0;
543  }
544 
545  // validate duty cycle limits, both limits must be between
546  // 0.0 and 1.0 (inclusive) and max must be greater then min
547  if (*(hd->dcm_max_dc) > 1.0) {
548  *(hd->dcm_max_dc) = 1.0;
549  }
550  if (*(hd->dcm_min_dc) > *(hd->dcm_max_dc)) {
551  *(hd->dcm_min_dc) = *(hd->dcm_max_dc);
552  }
553  if (*(hd->dcm_min_dc) < -1.0) {
554  *(hd->dcm_min_dc) = -1.0;
555  }
556  if (*(hd->dcm_max_dc) < *(hd->dcm_min_dc)) {
557  *(hd->dcm_max_dc) = *(hd->dcm_min_dc);
558  }
559 
560  // do scale calcs only when scale changes
561  if (*(hd->dcm_scale) != hd->dcm_old_scale) {
562  // validate the new scale value
563  if ((*(hd->dcm_scale) < 1e-20) && (*(hd->dcm_scale) > -1e-20)) {
564  // value too small, divide by zero is a bad thing
565  *(hd->dcm_scale) = 1.0;
566  }
567  // get ready to detect future scale changes
568  hd->dcm_old_scale = *(hd->dcm_scale);
569  // we will need the reciprocal
570  hd->dcm_scale_recip = 1.0 / *(hd->dcm_scale);
571  }
572 
573  // get command
574  tmpval = *(hd->dcm_value);
575  if (*(hd->dcm_absmode) && (tmpval < 0)) {
576  tmpval = -tmpval;
577  }
578 
579  // convert value command to duty cycle
580  tmpdc = tmpval * hd->dcm_scale_recip + *(hd->dcm_offset);
581  if (tmpdc < *(hd->dcm_min_dc)) {
582  tmpdc = *(hd->dcm_min_dc);
583  }
584  if (tmpdc > *(hd->dcm_max_dc)) {
585  tmpdc = *(hd->dcm_max_dc);
586  }
587 
588  // set output values
589  if (*(hd->dcm_enable) == 0) {
590  raw_val = 0;
591  *(hd->dcm_curr_dc) = 0;
592  } else {
593  raw_val = (double)0x7fff * tmpdc;
594  if (raw_val > (double)0x7fff) {
595  raw_val = (double)0x7fff;
596  }
597  if (raw_val < (double)-0x7fff) {
598  raw_val = (double)-0x7fff;
599  }
600  *(hd->dcm_curr_dc) = tmpdc;
601  }
602 
603  // update value
604  *(hd->dcm_raw_val) = (int32_t)raw_val;
605 
606  // fault reset
607  if (*(hd->fault_reset)) {
608  *(hd->dcm_reset) = 1;
609  }
610 
611  if (hd->fault_reset_retry > 0) {
612  if (hd->fault_reset_state) {
613  *(hd->dcm_reset) = 1;
614  }
615  } else {
616  if (*(hd->dcm_enable)) {
617  *(hd->dcm_reset) = 0;
618  }
619  *(hd->fault_reset) = 0;
620  }
621 
622  // set output data
623  EC_WRITE_BIT(&pd[hd->set_count_pdo_os], hd->set_count_pdo_bp, *(hd->set_raw_count));
624  EC_WRITE_BIT(&pd[hd->ena_latch_c_pdo_os], hd->ena_latch_c_pdo_bp, *(hd->ena_latch_c));
625  EC_WRITE_BIT(&pd[hd->ena_latch_ext_pos_pdo_os], hd->ena_latch_ext_pos_pdo_bp, *(hd->ena_latch_ext_pos));
626  EC_WRITE_BIT(&pd[hd->ena_latch_ext_neg_pdo_os], hd->ena_latch_ext_neg_pdo_bp, *(hd->ena_latch_ext_neg));
627  EC_WRITE_S16(&pd[hd->set_count_val_pdo_os], *(hd->set_raw_count_val));
628 
629  EC_WRITE_BIT(&pd[hd->dcm_ena_pdo_os], hd->dcm_ena_pdo_bp, *(hd->dcm_enable));
630  EC_WRITE_BIT(&pd[hd->dcm_reset_pdo_os], hd->dcm_reset_pdo_bp, *(hd->dcm_reset));
631  EC_WRITE_BIT(&pd[hd->dcm_reduce_torque_pdo_os], hd->dcm_reduce_torque_pdo_bp, *(hd->dcm_reduce_torque));
632  EC_WRITE_S16(&pd[hd->dcm_velo_pdo_os], (int16_t)raw_val);
