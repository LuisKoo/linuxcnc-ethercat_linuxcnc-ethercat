444 static void lcec_el3xxx_read(struct lcec_slave *slave, long period, uint32_t mask, int has_sync) {
445  lcec_master_t *master = slave->master;
446  lcec_el3xxx_data_t *hal_data = (lcec_el3xxx_data_t *)slave->hal_data;

448  int i;
449  lcec_el3xxx_chan_t *chan;
450  int16_t value;
451 
452  // wait for slave to be operational
453  if (!slave->state.operational) {
454  return;
455  }
456 
457  // check inputs
458  for (i = 0; i < hal_data->channels; i++) {
459  chan = &hal_data->chans[i];
460 
461  // update state
462  // update state
463  *(chan->overrange) = EC_READ_BIT(&pd[chan->ovr_pdo_os], chan->ovr_pdo_bp);
464  *(chan->underrange) = EC_READ_BIT(&pd[chan->udr_pdo_os], chan->udr_pdo_bp);
465  *(chan->error) = EC_READ_BIT(&pd[chan->error_pdo_os], chan->error_pdo_bp);
466  if (has_sync) *(chan->sync_err) = EC_READ_BIT(&pd[chan->sync_err_pdo_os], chan->sync_err_pdo_bp);
467 
468  // update value
469  value = EC_READ_S16(&pd[chan->val_pdo_os]) & mask;
470  *(chan->raw_val) = value;
471  *(chan->val) = *(chan->bias) + *(chan->scale) * (double)value * ((double)1 / (double)mask);
472  }
473 }
