105 static int lcec_el41x4_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
106  lcec_master_t *master = slave->master;
107  lcec_el41x4_data_t *hal_data;
108  lcec_el41x4_chan_t *chan;
109  int i;
110  int err;
111 
112  // initialize callbacks
113  slave->proc_write = lcec_el41x4_write;
114 
115  // alloc hal memory
116  if ((hal_data = hal_malloc(sizeof(lcec_el41x4_data_t))) == NULL) {
117  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
118  return -EIO;
119  }
120  memset(hal_data, 0, sizeof(lcec_el41x4_data_t));
121  slave->hal_data = hal_data;
122 
123  // initializer sync info
124  slave->sync_info = lcec_el41x4_syncs;
125 
126  // initialize pins
127  for (i=0; i<LCEC_EL41x4_CHANS; i++) {
128  chan = &hal_data->chans[i];
129 
130  // initialize POD entries
131  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7000 + (i*0x10), 0x01, &chan->val_pdo_os, NULL);
132 
133  // export pins
134  if ((err = lcec_pin_newf_list(chan, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
135  return err;
136  }
137 
138  // set default pin values
139  *(chan->scale) = 1.0;
140  *(chan->min_dc) = -1.0;
141  *(chan->max_dc) = 1.0;
142 
143  // init other fields
144  chan->old_scale = *(chan->scale) + 1.0;
145  chan->scale_recip = 1.0;
146  }
147 
148  return 0;
149 }

151 static void lcec_el41x4_write(struct lcec_slave *slave, long period) {
152  lcec_master_t *master = slave->master;
153  lcec_el41x4_data_t *hal_data = (lcec_el41x4_data_t *) slave->hal_data;
154  uint8_t *pd = master->process_data;
155  int i;
156  lcec_el41x4_chan_t *chan;
157  double tmpval, tmpdc, raw_val;
158 
159  // set outputs
160  for (i=0; i<LCEC_EL41x4_CHANS; i++) {
161  chan = &hal_data->chans[i];
162 
163  // validate duty cycle limits, both limits must be between
164  // 0.0 and 1.0 (inclusive) and max must be greater then min
165  if (*(chan->max_dc) > 1.0) {
166  *(chan->max_dc) = 1.0;
167  }
168  if (*(chan->min_dc) > *(chan->max_dc)) {
169  *(chan->min_dc) = *(chan->max_dc);
170  }
171  if (*(chan->min_dc) < -1.0) {
172  *(chan->min_dc) = -1.0;
173  }
174  if (*(chan->max_dc) < *(chan->min_dc)) {
175  *(chan->max_dc) = *(chan->min_dc);
176  }
177 
178  // do scale calcs only when scale changes
179  if (*(chan->scale) != chan->old_scale) {
180  // validate the new scale value
181  if ((*(chan->scale) < 1e-20) && (*(chan->scale) > -1e-20)) {
182  // value too small, divide by zero is a bad thing
183  *(chan->scale) = 1.0;
184  }
185  // get ready to detect future scale changes
186  chan->old_scale = *(chan->scale);
187  // we will need the reciprocal
188  chan->scale_recip = 1.0 / *(chan->scale);
189  }
190 
191  // get command
192  tmpval = *(chan->value);
193  if (*(chan->absmode) && (tmpval < 0)) {
194  tmpval = -tmpval;
195  }
196 
197  // convert value command to duty cycle
198  tmpdc = tmpval * chan->scale_recip + *(chan->offset);
199  if (tmpdc < *(chan->min_dc)) {
200  tmpdc = *(chan->min_dc);
201  }
202  if (tmpdc > *(chan->max_dc)) {
203  tmpdc = *(chan->max_dc);
204  }
205 
206  // set output values
207  if (*(chan->enable) == 0) {
208  raw_val = 0;
209  *(chan->pos) = 0;
210  *(chan->neg) = 0;
211  *(chan->curr_dc) = 0;
212  } else {
213  raw_val = (double)0x7fff * tmpdc;
214  if (raw_val > (double)0x7fff) {
215  raw_val = (double)0x7fff;
216  }
217  if (raw_val < (double)-0x7fff) {
218  raw_val = (double)-0x7fff;
219  }
220  *(chan->pos) = (*(chan->value) > 0);
221  *(chan->neg) = (*(chan->value) < 0);
222  *(chan->curr_dc) = tmpdc;
223  }
224 
225  // update value
226  EC_WRITE_S16(&pd[chan->val_pdo_os], (int16_t)raw_val);
227  *(chan->raw_val) = (int32_t)raw_val;
228  }
