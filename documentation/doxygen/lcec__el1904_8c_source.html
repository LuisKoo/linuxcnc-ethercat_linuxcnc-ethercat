93 static int lcec_el1904_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
94  lcec_master_t *master = slave->master;
95  lcec_el1904_data_t *hal_data;
96  int i, err;
97  lcec_el1904_data_in_t *in;
98 
99  // initialize callbacks
100  slave->proc_read = lcec_el1904_read;
101 
102  // alloc hal memory
103  if ((hal_data = hal_malloc(sizeof(lcec_el1904_data_t))) == NULL) {
104  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
105  return -EIO;
106  }
107  memset(hal_data, 0, sizeof(lcec_el1904_data_t));
108  slave->hal_data = hal_data;
109 
110  // initialize POD entries
111  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7000, 0x01, &hal_data->fsoe_master_cmd_os, NULL);
112  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7000, 0x02, &hal_data->fsoe_master_crc_os, NULL);
113  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x7000, 0x03, &hal_data->fsoe_master_connid_os, NULL);
114  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000, 0x01, &hal_data->fsoe_slave_cmd_os, NULL);
115  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000, 0x03, &hal_data->fsoe_slave_crc_os, NULL);
116  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6000, 0x04, &hal_data->fsoe_slave_connid_os, NULL);
117  for (i = 0, in = hal_data->inputs; i < LCEC_EL1904_INPUT_COUNT; i++, in++) {
118  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6001, 0x01 + i, &in->fsoe_in_os, &in->fsoe_in_bp);
119  }
120 
121  // export pins
122  if ((err = lcec_pin_newf_list(hal_data, slave_pins, LCEC_MODULE_NAME, master->name, slave->name)) != 0) {
123  return err;
124  }
125  for (i = 0, in = hal_data->inputs; i < LCEC_EL1904_INPUT_COUNT; i++, in++) {
126  if ((err = lcec_pin_newf_list(in, slave_in_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
127  return err;
128  }
129  }
130 
131  return 0;
132 }

134 static void lcec_el1904_read(struct lcec_slave *slave, long period) {
135  lcec_master_t *master = slave->master;
136  lcec_el1904_data_t *hal_data = (lcec_el1904_data_t *) slave->hal_data;
137  uint8_t *pd = master->process_data;
138  int i;
139  lcec_el1904_data_in_t *in;
140 
141  copy_fsoe_data(slave, hal_data->fsoe_slave_cmd_os, hal_data->fsoe_master_cmd_os);
142 
143  *(hal_data->fsoe_slave_cmd) = EC_READ_U8(&pd[hal_data->fsoe_slave_cmd_os]);
144  *(hal_data->fsoe_slave_crc) = EC_READ_U16(&pd[hal_data->fsoe_slave_crc_os]);
145  *(hal_data->fsoe_slave_connid) = EC_READ_U16(&pd[hal_data->fsoe_slave_connid_os]);
146 
147  *(hal_data->fsoe_master_cmd) = EC_READ_U8(&pd[hal_data->fsoe_master_cmd_os]);
148  *(hal_data->fsoe_master_crc) = EC_READ_U16(&pd[hal_data->fsoe_master_crc_os]);
149  *(hal_data->fsoe_master_connid) = EC_READ_U16(&pd[hal_data->fsoe_master_connid_os]);
150 
151  for (i = 0, in = hal_data->inputs; i < LCEC_EL1904_INPUT_COUNT; i++, in++) {
152  *(in->fsoe_in) = EC_READ_BIT(&pd[in->fsoe_in_os], in->fsoe_in_bp);
153  *(in->fsoe_in_not) = ! *(in->fsoe_in);
154  }
155 }
