49 static int lcec_ex260_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
50  lcec_master_t *master = slave->master;
51  lcec_ex260_pin_t *hal_data;
52  lcec_ex260_pin_t *pin;
53  int i;
54  int err;
55 
56  // initialize callbacks
57  slave->proc_write = lcec_ex260_write;
58 
59  // alloc hal memory
60  if ((hal_data = hal_malloc(sizeof(lcec_ex260_pin_t) * slave->pdo_entry_count)) == NULL) {
61  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
62  return -EIO;
63  }
64  memset(hal_data, 0, sizeof(lcec_ex260_pin_t) * slave->pdo_entry_count);
65  slave->hal_data = hal_data;
66 
67  // initialize pins
68  for (i = 0, pin = hal_data; i < slave->pdo_entry_count; i++, pin++) {
69  // initialize PDO entry
70  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x3101, 0x01 + i, &pin->pdo_os, &pin->pdo_bp);
71 
72  if ((err = lcec_pin_newf_list(pin, slave_pins, LCEC_MODULE_NAME, master->name, slave->name, i)) != 0) {
73  return err;
74  }
75  }
76 
77  return 0;
78 }

80 static void lcec_ex260_write(struct lcec_slave *slave, long period) {
81  lcec_master_t *master = slave->master;
82  lcec_ex260_pin_t *hal_data = (lcec_ex260_pin_t *)slave->hal_data;
83  uint8_t *pd = master->process_data;
84  lcec_ex260_pin_t *pin;
85  int i, s;
86 
87  // wait for slave to be operational
88  if (!slave->state.operational) {
89  return;
90  }
91 
92  // set outputs
93  for (i = 0, pin = hal_data; i < slave->pdo_entry_count; i++, pin++) {
94  s = *(pin->sol_1a);

97  s |= *(pin->sol_2b) << 3;

99  s |= *(pin->sol_3b) << 5;
