211 static int lcec_omrg5_init(int comp_id, struct lcec_slave *slave, ec_pdo_entry_reg_t *pdo_entry_regs) {
212  lcec_master_t *master = slave->master;
213  lcec_omrg5_data_t *hal_data;
214  int err;
215 
216  // initialize callbacks
217  slave->proc_read = lcec_omrg5_read;
218  slave->proc_write = lcec_omrg5_write;
219 
220  // alloc hal memory
221  if ((hal_data = hal_malloc(sizeof(lcec_omrg5_data_t))) == NULL) {
222  rtapi_print_msg(RTAPI_MSG_ERR, LCEC_MSG_PFX "hal_malloc() for slave %s.%s failed\n", master->name, slave->name);
223  return -EIO;
224  }
225  memset(hal_data, 0, sizeof(lcec_omrg5_data_t));
226  slave->hal_data = hal_data;
227 
228  // set to cyclic synchronous position mode
229  if (ecrt_slave_config_sdo8(slave->config, 0x6060, 0x00, 8) != 0) {
230  rtapi_print_msg (RTAPI_MSG_ERR, LCEC_MSG_PFX "fail to configure slave %s.%s sdo velo mode\n", master->name, slave->name);
231  }
232 
233  // initialize sync info
234  slave->sync_info = lcec_omrg5_syncs;
235 
236  // initialize POD entries
237  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x603F, 0x00, &hal_data->error_pdo_os, NULL);
238  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6041, 0x00, &hal_data->status_pdo_os, NULL);
239  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6064, 0x00, &hal_data->curr_pos_pdo_os, NULL);
240  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6077, 0x00, &hal_data->curr_torque_pdo_os, NULL);
241  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60F4, 0x00, &hal_data->curr_ferr_pdo_os, NULL);
242  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60B9, 0x00, &hal_data->latch_stat_pdo_os, NULL);
243  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60BA, 0x00, &hal_data->latch_pos1_pdo_os, NULL);
244  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60BC, 0x00, &hal_data->latch_pos2_pdo_os, NULL);
245  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60FD, 0x00, &hal_data->din_pdo_os, NULL);
246  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x6040, 0x00, &hal_data->control_pdo_os, NULL);
247  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x607A, 0x00, &hal_data->target_pos_pdo_os, NULL);
248  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60B8, 0x00, &hal_data->latch_fnk_os, NULL);
249  LCEC_PDO_INIT(pdo_entry_regs, slave->index, slave->vid, slave->pid, 0x60FE, 0x01, &hal_data->dout_pdo_os, NULL);
250 
251  // export pins
252  if ((err = lcec_pin_newf_list(hal_data, slave_pins, LCEC_MODULE_NAME, master->name, slave->name)) != 0) {
253  return err;
254  }
255 
256  // export parameters
257  if ((err = lcec_param_newf_list(hal_data, slave_params, LCEC_MODULE_NAME, master->name, slave->name)) != 0) {
258  return err;
259  }
260 
261  // initialize variables
262  hal_data->pos_scale = (double) OMRG5_PULSES_PER_REV_DEFLT;
263  hal_data->pos_scale_old = hal_data->pos_scale + 1.0;
264  hal_data->pos_scale_rcpt = 1.0;
265  hal_data->auto_fault_reset = 1;
266 
267  return 0;
268 }

270 static void lcec_omrg5_check_scales(lcec_omrg5_data_t *hal_data) {
271  // check for change in scale value
272  if (hal_data->pos_scale != hal_data->pos_scale_old) {
273 
274  // scale value has changed, test and update it
275  if ((hal_data->pos_scale < 1e-20) && (hal_data->pos_scale > -1e-20)) {
276  // value too small, divide by zero is a bad thing
277  hal_data->pos_scale = 1.0;
278  }
279 
280  // save new scale to detect future changes
281  hal_data->pos_scale_old = hal_data->pos_scale;
282 
283  // we actually want the reciprocal
284  hal_data->pos_scale_rcpt = 1.0 / hal_data->pos_scale;
285  }
286 }

288 static void lcec_omrg5_read(struct lcec_slave *slave, long period) {
289  lcec_master_t *master = slave->master;
290  lcec_omrg5_data_t *hal_data = (lcec_omrg5_data_t *) slave->hal_data;
291  uint8_t *pd = master->process_data;
292  uint16_t status;
293  uint32_t din;
294 
295  // check for change in scale value
296  lcec_omrg5_check_scales(hal_data);
297 
298  // read status
299  status = EC_READ_U16(&pd[hal_data->status_pdo_os]);
300  *(hal_data->stat_switchon_ready) = (status >> 0) & 1;
301  *(hal_data->stat_switched_on) = (status >> 1) & 1;
302  *(hal_data->stat_op_enabled) = (status >> 2) & 1;
303  *(hal_data->stat_fault) = (status >> 3) & 1;
304  *(hal_data->stat_volt_enabled) = (status >> 4) & 1;
305  *(hal_data->stat_quick_stop) = (status >> 5) & 1;
306  *(hal_data->stat_switchon_disabled) = (status >> 6) & 1;
307  *(hal_data->stat_warning) = (status >> 7) & 1;
308  *(hal_data->stat_remote) = (status >> 9) & 1;
309 
310  // read digital inputs
311  din = EC_READ_U32(&pd[hal_data->din_pdo_os]);
312  *(hal_data->din_not) = (din >> 0) & 1;
313  *(hal_data->din_pot) = (din >> 1) & 1;
314  *(hal_data->din_dec) = (din >> 2) & 1;
315  *(hal_data->din_pc) = (din >> 16) & 1;
316  *(hal_data->din_ext1) = (din >> 17) & 1;
317  *(hal_data->din_ext2) = (din >> 18) & 1;
318  *(hal_data->din_ext3) = (din >> 19) & 1;
319  *(hal_data->din_mon0) = (din >> 20) & 1;
320  *(hal_data->din_mon1) = (din >> 21) & 1;
321  *(hal_data->din_mon2) = (din >> 22) & 1;
322  *(hal_data->din_pcl) = (din >> 23) & 1;
323  *(hal_data->din_ncl) = (din >> 24) & 1;
324  *(hal_data->din_stop) = (din >> 25) & 1;
325  *(hal_data->din_bkir) = (din >> 26) & 1;
326  *(hal_data->din_sf1) = (din >> 27) & 1;
327  *(hal_data->din_sf2) = (din >> 28) & 1;
328  *(hal_data->din_edm) = (din >> 29) & 1;
329 
330  // read position feedback
331  *(hal_data->pos_fb_raw) = EC_READ_S32(&pd[hal_data->curr_pos_pdo_os]);
332  *(hal_data->pos_fb) = ((double) *(hal_data->pos_fb_raw)) * hal_data->pos_scale_rcpt;
333 
334  // read following error
335  *(hal_data->pos_ferr_raw) = EC_READ_S32(&pd[hal_data->curr_ferr_pdo_os]);
336  *(hal_data->pos_ferr) = ((double) *(hal_data->pos_ferr_raw)) * hal_data->pos_scale_rcpt;
337 
338  // read torque error
339  *(hal_data->torque_fb) = ((double) EC_READ_S16(&pd[hal_data->curr_torque_pdo_os])) * 0.1;
340 
341  // read error code
342  *(hal_data->error_code) = EC_READ_U16(&pd[hal_data->error_pdo_os]);
343 
344  // update fault output
345  if (hal_data->auto_fault_reset_delay > 0) {
346  hal_data->auto_fault_reset_delay -= period;
347  *(hal_data->fault) = 0;
348  } else {
349  *(hal_data->fault) = *(hal_data->stat_fault) && *(hal_data->enable);
350  }
351 }

353 static void lcec_omrg5_write(struct lcec_slave *slave, long period) {
354  lcec_master_t *master = slave->master;
355  lcec_omrg5_data_t *hal_data = (lcec_omrg5_data_t *) slave->hal_data;
356  uint8_t *pd = master->process_data;
357  int enable_edge;
358  uint16_t control;
359 
360  // detect enable edge
361  enable_edge = *(hal_data->enable) && !hal_data->enable_old;
362  hal_data->enable_old = *(hal_data->enable);
363 
364  // write control register
365  control = (1 << 2); // quick stop is not supported
366  if (*(hal_data->stat_fault)) {
367  if (*(hal_data->fault_reset)) {
368  control |= (1 << 7); // fault reset
369  }
370  if (hal_data->auto_fault_reset && enable_edge) {
371  hal_data->auto_fault_reset_delay = OMRG5_FAULT_AUTORESET_DELAY_NS;
372  control |= (1 << 7); // fault reset
373  }
374  } else {
375  if (*(hal_data->enable)) {
376  control |= (1 << 1); // enable voltage
377  if (*(hal_data->stat_switchon_ready)) {
378  control |= (1 << 0); // switch on
379  if (*(hal_data->stat_switched_on)) {
380  control |= (1 << 3); // enable op
381  }
382  }
383  }
384  }
385  EC_WRITE_U16(&pd[hal_data->control_pdo_os], control);
386 
387  // write position command
388  *(hal_data->pos_cmd_raw) = (int32_t) (*(hal_data->pos_cmd) * hal_data->pos_scale);
389  EC_WRITE_S32(&pd[hal_data->target_pos_pdo_os], *(hal_data->pos_cmd_raw));
390 }
